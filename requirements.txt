# v9.0 -- pinned versions for the "fat" build (everything baked into the exe
# at build time, nothing pip-installed on the user's machine at runtime).
#
# transformers is pinned BELOW 5.0 on purpose: jina-embeddings-v3 ships a
# custom trust_remote_code modeling file (XLMRobertaLoRA) written against
# the transformers v4.x tied-weights API. transformers v5 changed that
# internal API, so the custom code crashes with:
#   'XLMRobertaLoRA' object has no attribute 'all_tied_weights_keys'
# 4.44+ is required for proper Gemma2 support (bge_gemma2).
transformers>=4.44,<5

sentence-transformers>=3.0
einops
huggingface_hub

# BM25 keyword search — separate from the AI/semantic stack above, but was

rank_bm25

# Vietnamese diacritics restoration (v9.4) -- yammdd/vietnamese-error-correction
# is a LoRA adapter (needs peft) on top of vinai/bartpho-syllable (needs
# sentencepiece for its tokenizer). Baked into _internal at build time via
# scripts/download_vi_diacritics_model.py + the .spec file's `datas` entry,
# not downloaded at runtime.
# OCR for images (.jpg/.png/...) — optional, off by default (Update DB
# "OCR images" checkbox). EasyOCR chosen over Tesseract because it needs
# no separate native binary installer, and has solid built-in support for
# Japanese + Vietnamese + English together.
easyocr

sentencepiece
peft

# CPU-only torch build. This is dramatically smaller than the default
# CUDA-enabled wheel (which bundles multiple GB of CUDA/cuDNN binaries) and
# is the right choice unless you specifically need GPU inference on the
# target machines. If you DO want CUDA support, remove the --extra-index-url
# line below and pin a matching torch version + CUDA toolkit instead --
# talk to Claude about this before changing it, since it affects the whole
# build pipeline size and the .spec file's collected binaries.
--extra-index-url https://download.pytorch.org/whl/cpu
torch
